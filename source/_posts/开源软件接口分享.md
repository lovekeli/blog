<script>
    function getRootPath(url) {
        if (!url) {
            url = window.location.href;
        }
        const parser = document.createElement('a');
        parser.href = url;
        let rootPath = parser.protocol + '//' + parser.hostname;
        if (parser.port) {
            rootPath += ':' + parser.port;
        }
        return rootPath;
    }
    function copyInterface(str) {
        console.log(getRootPath()+str);
        var aux = document.createElement("input");
        aux.setAttribute("value", getRootPath()+str);
        document.body.appendChild(aux);
        aux.select();
        document.execCommand("copy");
        document.body.removeChild(aux);
        alert("复制成功");
    }
</script>
